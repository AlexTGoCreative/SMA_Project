## Install Backend Dependencies
cd "C:\Users\alexandru.tulbure\Documents\try\backend"
npm install

## Create .env file in backend folder
# Copy env.example to .env:
cd "C:\Users\alexandru.tulbure\Documents\try\backend"
copy env.example .env

## Start Backend Server (Terminal 1)
cd "C:\Users\alexandru.tulbure\Documents\try\backend"
node server.js

## Start Expo App (Terminal 2)
cd "C:\Users\alexandru.tulbure\Documents\try"
npx expo start --tunnel

## Find Your IP Address (to update in utils/api.js)
ipconfig

## Start MongoDB (if not running)
net start MongoDB

## Allow Node through Firewall (if needed)
netsh advfirewall firewall add rule name="Node Server" dir=in action=allow protocol=TCP localport=3000
